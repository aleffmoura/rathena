-	script	mapa	-1,{

OnWhisperGlobal:

Set @mapa,10000000; // Zeny Cobrado de Player free Para Mandar Mensagem para o mapa

If (Zeny < @mapa) {
	mes "Cada Mensagem custa "+@mapa+"z";
	mes "você não tem o zeny necessario para mandar a mensagem";
	Close;
}
	set zeny,zeny-@mapa;
getmapxy .@mapname$, .@x, .@y, 0;
mapannounce .@mapname$,strcharinfo(0)+":"+@whispervar0$, bc_map;
dispbottom "Cada mensagem tem um custo de "+@mapa+"z";

end;
}
-	script	global	-1,{

OnWhisperGlobal:

Set @global,50000000; //Zeny Cobrado de player free para mandar Mensagem para todo servidor

If (Zeny < @global) {
	mes "Cada Mensagem custa "+@global+"z";
	mes "você não tem o zeny necessario para mandar a mensagem";
	Close;
}
Set Zeny, Zeny - @global;
announce ""+strcharinfo(0)+" : "+@whispervar0$+"",bc_yellow|bc_all;
dispbottom "Cada mensagem tem um custo de "+@global+"z";
end;
}
